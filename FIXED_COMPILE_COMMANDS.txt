### 2. If still errors, use this SIMPLIFIED process_utils.cpp:

Create a new file `process_utils_simple.cpp` with just basic functionality:

```cpp
#define _WIN32_WINNT 0x0600
#define WIN32_LEAN_AND_MEAN
#include <windows.h>
#include <tlhelp32.h>
#include <string>
#include <vector>
#include <sstream>

struct ProcessInfo {
    DWORD pid;
    std::string name;
    DWORD sessionId;
    SIZE_T memoryUsage;
    DWORD threadCount;
    DWORD parentPid;
    std::string architecture;
    std::string user;
    std::string path;
};

std::vector<ProcessInfo> GetProcessList() {
    std::vector<ProcessInfo> processes;
    HANDLE hSnapshot = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
    if (hSnapshot == INVALID_HANDLE_VALUE) return processes;
    
    PROCESSENTRY32 pe32;
    pe32.dwSize = sizeof(PROCESSENTRY32);
    
    if (!Process32First(hSnapshot, &pe32)) {
        CloseHandle(hSnapshot);
        return processes;
    }
    
    do {
        ProcessInfo info;
        info.pid = pe32.th32ProcessID;
        info.name = pe32.szExeFile;
        info.parentPid = pe32.th32ParentProcessID;
        info.threadCount = pe32.cntThreads;
        info.sessionId = 0;
        info.memoryUsage = 0;
        info.architecture = "Unknown";
        info.user = "Unknown";
        info.path = "";
        
        processes.push_back(info);
    } while (Process32Next(hSnapshot, &pe32));
    
    CloseHandle(hSnapshot);
    return processes;
}

std::string GetProcessListJSON() {
    auto processes = GetProcessList();
    std::ostringstream json;
    json << "[";
    bool first = true;
    
    for (const auto& proc : processes) {
        if (!first) json << ",";
        first = false;
        json << "{\"pid\":" << proc.pid << ",\"name\":\"" << proc.name << "\"}";
    }
    
    json << "]";
    return json.str();
}
```
